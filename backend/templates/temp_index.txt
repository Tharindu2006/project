    <div class="modal-content">
      <span class="close-modal" onclick="closeLogin()">&times;</span>
      <h2 id="modalTitle" style="margin-top:0; color:#004b8d;">Login</h2>
      <div id="authForm">
        <label>Email <input id="email" placeholder="name@example.com"></label>

        <div id="registerFields" style="display:none;">
          <label>Name <input id="name" placeholder="Your Name"></label>

            <select id="role">
              <option value="seeker">Person in need</option>
              <option value="caregiver">Caregiver</option>

            <button id="btnRegister" onclick="register()" style="flex:1; background-color:#2b6cb0; display:none;">Register</button>
        </div>

        <div id="me" style="font-weight:bold; margin-bottom:10px;">Not logged in</div>

  <script>
    let currentUser = null;
    const API_BASE = new URLSearchParams(location.search).get('api') || (window.API_BASE || '');

    const api = (path, opts={}) =>
      fetch(`${API_BASE}${path}`, Object.assign({credentials:"include", headers:{"Content-Type":"application/json"}}, opts));

    function setAuthenticatedUI(user) {
      const isAuthed = !!user;
      const reqSection = document.getElementById('requests-section');
      const cgSection = document.getElementById('caregivers-section');
      [reqSection, cgSection].forEach(el => el.classList.toggle('hidden', !isAuthed));

      // Hide navigation links and swap login/avatar based on auth
      document.querySelectorAll('.nav-links a').forEach(link => link.classList.toggle('hidden', !isAuthed));
      const loginLink = document.getElementById('loginLink');
      const avatar = document.getElementById('userAvatar');
      loginLink.classList.toggle('hidden', isAuthed);
      if (isAuthed) {
        const initial = (user.name || user.email || '?').trim().charAt(0).toUpperCase() || '?';
        avatar.textContent = initial;
        avatar.classList.remove('hidden');
      } else {
        avatar.textContent = '';
        avatar.classList.add('hidden');
      }
    }

    function openLogin() {
      document.getElementById('loginModal').style.display = 'block';
      document.getElementById('modalTitle').innerText = 'Login';
      document.getElementById('registerFields').style.display = 'none';
      document.getElementById('btnLogin').style.display = 'block';
      document.getElementById('btnRegister').style.display = 'none';
      document.getElementById('email').focus();
    }

    function openRegister(role) {
      document.getElementById('loginModal').style.display = 'block';
      document.getElementById('modalTitle').innerText = 'Register';
      document.getElementById('registerFields').style.display = 'block';
      document.getElementById('btnLogin').style.display = 'none';
      document.getElementById('btnRegister').style.display = 'block';
      if(role) document.getElementById('role').value = role;
      document.getElementById('email').focus();
    }

    function closeLogin() {
      document.getElementById('loginModal').style.display = 'none';
    }

    window.onclick = function(event) {
      if (event.target == document.getElementById('loginModal')) {
        closeLogin();
      }
    }

    async function fetchMe() {
      const res = await api('/api/me');
      const data = await res.json();
      currentUser = data.user;
      me.textContent = currentUser ? `Logged in as ${currentUser.name} (${currentUser.role})` : 'Not logged in';
      setAuthenticatedUI(currentUser);
      toggleForms(currentUser);
    }

    async function initHospitals() {
      const res = await api('/api/hospitals');
      if (!res.ok) {
        const msg = (await res.json())?.error || res.statusText;
        alert(`Failed to load hospitals: ${msg}`);
        return;
      }
      const hospitals = await res.json();
      for (const id of ['hospitalSelect','requestHospitalFilter','caregiverHospitalFilter','reqHospital']) {
        const el = document.getElementById(id);
        el.innerHTML = '<option value=\"\">All hospitals</option>';
        hospitals.forEach(h => {
          const o = document.createElement('option');
          o.value = h.id; o.textContent = h.name;
          el.appendChild(o.cloneNode(true));
        });
      }
      // keep reqHospital single-select without "all"
      document.getElementById('reqHospital').remove(0);
    }

    const val = (id) => document.getElementById(id).value.trim();

    async function register() {
      const hospital_ids = Array.from(document.getElementById('hospitalSelect').selectedOptions).map(o=>Number(o.value));
      const body = {
        email: val('email'),
        password: val('password'),
        name: val('name'),
        role: document.getElementById('role').value,
        hospital_ids,
      };
      const res = await api('/api/register',{method:'POST',body:JSON.stringify(body)});
      const data = await res.json();
      currentUser = res.ok ? data.user : null;
      me.textContent = res.ok ? `Logged in as ${data.user.name} (${data.user.role})` : data.error;
      if (res.ok) closeLogin();
      setAuthenticatedUI(currentUser);
      toggleForms(currentUser);
      loadRequests(); loadCaregivers();
    }

    async function login() {
      const res = await api('/api/login',{method:'POST',body:JSON.stringify({email:val('email'),password:val('password')})});
      const data = await res.json();
      currentUser = res.ok ? data.user : null;
      me.textContent = res.ok ? `Logged in as ${data.user.name} (${data.user.role})` : data.error;
      if (res.ok) closeLogin();
      setAuthenticatedUI(currentUser);
      toggleForms(currentUser);
      loadRequests(); loadCaregivers();
    }

    async function logout() {
      await api('/api/logout',{method:'POST'});
      currentUser = null;
      me.textContent = 'Not logged in';
      setAuthenticatedUI(null);
      toggleForms(null);
      loadRequests(); loadCaregivers();
    }

    function toggleForms(user) {
      document.getElementById('requestForm').style.display = user && user.role==='seeker' ? 'block':'none';
      document.getElementById('logoutBtn').disabled = !user;
    }

    async function createRequest() {
      const body = { title:reqTitle.value, description:reqDesc.value, hospital_id:Number(reqHospital.value) };
      const res = await api('/api/care-requests',{method:'POST',body:JSON.stringify(body)});
      const data = await res.json();
      alert(res.ok ? 'Posted!' : data.error);
      loadRequests();
    }

    async function loadRequests() {
      if (!currentUser) {
        requests.innerHTML = '<div class="card">Login to view requests.</div>';
        return;
      }
      const hospital_id = requestHospitalFilter.value;
      const res = await api(`/api/care-requests${hospital_id?`?hospital_id=${hospital_id}`:''}`);
      const items = await res.json();
      requests.innerHTML = items.map(r => {
        const accepted = r.accepted_by || [];
        const acceptedHtml = accepted.length
          ? accepted.map(a => {
              const cg = a.caregiver;
              const canAct = currentUser && currentUser.role === 'seeker' && currentUser.id === r.seeker.id;
              const actionBtn = !canAct
                ? ''
                : a.status === 'rejected'
                  ? `<button onclick="approve(${r.id}, ${a.acceptance_id})">Approve</button>`
                  : `<button onclick="reject(${r.id}, ${a.acceptance_id})">Reject</button>`;
              return `<div class="pill">${cg.name} (${a.status})</div>${actionBtn}`;
            }).join('')
          : '<span class="pill">No acceptances yet</span>';
        return `
          <div class="card">
            <div><strong>${r.title}</strong></div>
            <div>${r.description}</div>
            <div><span class="pill">${r.hospital.name}</span> status: ${r.status}</div>
            ${currentUser && currentUser.role === 'caregiver' ? `<button onclick="accept(${r.id})">Accept</button>` : ''}
            <div>Accepted by: ${acceptedHtml}</div>
          </div>
        `;
      }).join('');
    }

    async function accept(id) {
      const res = await api(`/api/care-requests/${id}/accept`,{method:'POST'});
      const data = await res.json();
      alert(res.ok ? 'Accepted' : data.error);
      loadRequests();
    }

    async function reject(requestId, acceptanceId) {
      const res = await api(`/api/care-requests/${requestId}/reject`,{
        method:'POST',
        body: JSON.stringify({ acceptance_id: acceptanceId })
      });
      const data = await res.json();
      alert(res.ok ? 'Rejected' : data.error);
      loadRequests();
    }

    async function approve(requestId, acceptanceId) {
      const res = await api(`/api/care-requests/${requestId}/approve`,{
        method:'POST',
        body: JSON.stringify({ acceptance_id: acceptanceId })
      });
      const data = await res.json();
      alert(res.ok ? 'Approved' : data.error);
      loadRequests();
    }

    async function loadCaregivers() {
      if (!currentUser) {
        caregivers.innerHTML = '<div class="card">Login to view caregivers.</div>';
        return;
      }
      const hospital_id = caregiverHospitalFilter.value;
      const res = await api(`/api/caregivers${hospital_id?`?hospital_id=${hospital_id}`:''}`);
      const items = await res.json();
      caregivers.innerHTML = items.map(c => `
        <div class="card">
          <div><strong>${c.name}</strong> <span class="pill">${c.role}</span></div>
          <div>${c.bio || ''}</div>
          <div>${(c.hospitals||[]).map(h=>`<span class="pill">${h.name}</span>`).join('')}</div>
          <div>${c.phone||''}</div>
        </div>
      `).join('');
    }

    initHospitals().then(()=>{ fetchMe().then(()=>{ loadRequests(); loadCaregivers(); }); });
  </script>
